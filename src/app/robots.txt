import type { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        // Block canvas and demo from indexing — they're app pages not content
        disallow: ['/canvas', '/demo'],
      },
    ],
    sitemap: 'https://gridnoteapp.com/sitemap.xml',
  };
}
